### Dashboard API

```typescript
// Dashboard data structure
interface DashboardBriefing {
  greeting: string;
  clientSummary: {
    total: number;
    healthy: number;
    needsAttention: number;
  };
  todaysSessions: SessionBriefing[];
  needsAttention: AttentionItem[];
  celebrations: CelebrationItem[];
  activitySummary: {
    workoutsCompleted: number;
    workoutsRemaining: number;
    nutritionLogged: number;
    nutritionExpected: number;
  };
}

interface SessionBriefing {
  time: string;
  client: Client;
  sessionType: string;
  location: string;
  clientStatus: 'thriving' | 'steady' | 'struggling';
  recentHighlights: string[];
  patSuggestions: string[];
  todaysWorkout: WorkoutPlan;
}

interface AttentionItem {
  client: Client;
  urgency: 'urgent' | 'attention' | 'monitor';
  observation: string;
  clientSaid?: string;
  pattern?: string;
  suggestion: string;
  actions: Action[];
}

interface CelebrationItem {
  client: Client;
  achievement: string;
  details: string;
  patAlreadyDid?: string;
  suggestedAction: string;
}
```
