# HiPat Lead Generation Framework (Pluggable)

## Overview

This document specifies a **pluggable framework** for lead generation tools. HiPat provides the structure, UI shell, and data handling - but the actual tools (TDEE Calculator, Workout Assessment, etc.) are **external integrations** that can be swapped in.

## Data Model

### LeadGenTool Configuration

```typescript
interface LeadGenTool {
  id: string;
  
  // Display
  name: string;                    // "TDEE Calculator"
  description: string;             // "Nutrition focused assessment"
  icon: string;                    // Material symbol name
  color?: string;                  // Brand color
  
  // Integration
  type: 'iframe' | 'redirect' | 'component';
  
  // URLs for different modes
  liveUrl?: string;                // URL for "Start Live" mode
  selfServiceUrl?: string;         // URL for "Get Link" mode (shareable)
  
  // For iframe/redirect integrations
  embedUrl?: string;               // Base URL to embed/open
  
  // For component integrations (future - React component)
  componentName?: string;          // Registered component name
  
  // Data contract
  inputSchema?: object;            // What data to send TO the tool
  outputSchema?: object;           // What data to expect FROM the tool
  
  // Callbacks
  callbackMethod: 'postMessage' | 'webhook' | 'redirect';
  webhookUrl?: string;             // If using webhook callback
  redirectParam?: string;          // URL param name for redirect callback
  
  // Status
  isActive: boolean;
  isConfigured: boolean;           // Has valid URLs set
  
  // Mentor-specific overrides
  mentorOverrides?: {
    [mentorId: string]: {
      liveUrl?: string;
      selfServiceUrl?: string;
      customParams?: Record<string, string>;
    }
  };
  
  createdAt: Date;
  updatedAt: Date;
}
```

### Tool Instance (Per Mentor)

```typescript
interface MentorToolConfig {
  id: string;
  mentorId: string;
  toolId: string;
  
  // Override URLs (if mentor has own hosted version)
  liveUrlOverride?: string;
  selfServiceUrlOverride?: string;
  
  // Customization for self-service page
  headline?: string;
  subheadline?: string;
  showMentorProfile: boolean;
  showWorkWithMeCTA: boolean;
  
  // Tracking
  isEnabled: boolean;
  viewCount: number;
  completionCount: number;
  signupCount: number;
  
  createdAt: Date;
}
```

### Tool Submission (Results from external tool)

```typescript
interface ToolSubmission {
  id: string;
  toolId: string;
  mentorId: string;
  
  // Client info (from tool)
  clientData: {
    name?: string;
    email: string;
    phone?: string;
  };
  
  // Tool-specific results (flexible JSON)
  results: Record<string, any>;
  
  // Example TDEE results:
  // {
  //   bmr: 1842,
  //   tdee: 2856,
  //   dailyTarget: 3056,
  //   macros: { protein: 185, carbs: 340, fat: 85 },
  //   goal: 'build_muscle',
  //   activityLevel: 'moderate'
  // }
  
  // Example Workout results:
  // {
  //   experienceLevel: 'intermediate',
  //   daysPerWeek: 4,
  //   goal: 'strength',
  //   recommendedProgram: 'upper_lower_4day',
  //   currentLifts: { squat: 185, bench: 135, deadlift: 225 }
  // }
  
  // Conversion tracking
  status: 'submitted' | 'invited' | 'signed_up' | 'became_client';
  
  // Created entities
  clientId?: string;               // If added to roster
  patUserId?: string;              // If signed up for Pat
  
  // Timestamps
  submittedAt: Date;
  invitedAt?: Date;
  signedUpAt?: Date;
}
```

### 3. Tool Loader Component

Handles loading external tools in different modes:

```typescript
interface ToolLoaderProps {
  tool: LeadGenTool;
  mode: 'live' | 'self-service';
  mentorId: string;
  onComplete: (submission: ToolSubmission) => void;
  onCancel: () => void;
}
```

#### Iframe Mode

```tsx
function ToolLoaderIframe({ tool, mode, mentorId, onComplete, onCancel }) {
  const iframeRef = useRef<HTMLIFrameElement>(null);
  
  // Build URL with params
  const toolUrl = useMemo(() => {
    const baseUrl = mode === 'live' ? tool.liveUrl : tool.selfServiceUrl;
    const url = new URL(baseUrl);
    url.searchParams.set('mentorId', mentorId);
    url.searchParams.set('mode', mode);
    url.searchParams.set('callback', 'postMessage');
    return url.toString();
  }, [tool, mode, mentorId]);
  
  // Listen for postMessage from tool
  useEffect(() => {
    const handleMessage = (event: MessageEvent) => {
      // Validate origin matches tool URL
      if (!tool.liveUrl?.includes(event.origin)) return;
      
      if (event.data.type === 'TOOL_COMPLETE') {
        onComplete({
          toolId: tool.id,
          mentorId,
          clientData: event.data.clientData,
          results: event.data.results,
          status: 'submitted',
          submittedAt: new Date(),
        });
      }
      
      if (event.data.type === 'TOOL_CANCEL') {
        onCancel();
      }
    };
    
    window.addEventListener('message', handleMessage);
    return () => window.removeEventListener('message', handleMessage);
  }, [tool, mentorId, onComplete, onCancel]);
  
  return (
    <div className="tool-loader-container">
      <iframe
        ref={iframeRef}
        src={toolUrl}
        className="w-full h-[600px] border-0 rounded-lg"
        allow="clipboard-write"
      />
    </div>
  );
}
```

#### Redirect Mode

```tsx
function ToolLoaderRedirect({ tool, mode, mentorId }) {
  const callbackUrl = `${window.location.origin}/api/tool-callback/${tool.id}`;
  
  const handleLaunch = () => {
    const baseUrl = mode === 'live' ? tool.liveUrl : tool.selfServiceUrl;
    const url = new URL(baseUrl);
    url.searchParams.set('mentorId', mentorId);
    url.searchParams.set('mode', mode);
    url.searchParams.set('callbackUrl', callbackUrl);
    
    window.open(url.toString(), '_blank');
  };
  
  return (
    <div className="text-center p-8">
      <p className="mb-4">This tool will open in a new window.</p>
      <Button onClick={handleLaunch}>
        Open {tool.name}
      </Button>
      <p className="text-sm text-muted-foreground mt-4">
        Results will be sent back automatically when complete.
      </p>
    </div>
  );
}
```

## External Tool Contract

### What HiPat Sends TO the Tool

Via URL params when loading:

```
https://your-tool.com/calculator
  ?mentorId=mentor-123
  &mentorName=Coach%20Alex
  &mode=live|self-service
  &callback=postMessage|redirect
  &callbackUrl=https://hipat.app/api/tool-callback/tool-123
  &theme=light|dark
```

### What the Tool Sends BACK to HiPat

#### Option A: postMessage (for iframe)

```typescript
// When user completes the tool
window.parent.postMessage({
  type: 'TOOL_COMPLETE',
  clientData: {
    name: 'John Smith',
    email: 'john@email.com',
    phone: '555-123-4567'  // optional
  },
  results: {
    // Tool-specific results - flexible JSON
    // TDEE example:
    bmr: 1842,
    tdee: 2856,
    dailyTarget: 3056,
    macros: { protein: 185, carbs: 340, fat: 85 },
    goal: 'build_muscle',
    
    // Or Workout example:
    experienceLevel: 'intermediate',
    daysPerWeek: 4,
    recommendedProgram: 'upper_lower_4day',
    currentLifts: { squat: 185, bench: 135, deadlift: 225 }
  }
}, '*');

// When user cancels
window.parent.postMessage({
  type: 'TOOL_CANCEL'
}, '*');

// To resize iframe (optional)
window.parent.postMessage({
  type: 'TOOL_RESIZE',
  height: 800
}, '*');
```

#### Option B: Webhook (for redirect)

```typescript
// POST to callbackUrl
POST https://hipat.app/api/tool-callback/tool-123

{
  "mentorId": "mentor-123",
  "sessionId": "session-abc",  // Tracking ID from URL params
  "clientData": {
    "name": "John Smith",
    "email": "john@email.com"
  },
  "results": {
    // Same flexible structure
  }
}
```

#### Option C: Redirect with params

```
Redirect to:
https://hipat.app/tool-complete
  ?toolId=tool-123
  &mentorId=mentor-123
  &email=john@email.com
  &name=John%20Smith
  &results=BASE64_ENCODED_JSON
```

## HiPat Receives Results â†’ Creates Client

When HiPat receives tool completion data:

```typescript
async function handleToolCompletion(submission: ToolSubmission) {
  // 1. Store the submission
  const saved = await toolSubmissionsService.create(submission);
  
  // 2. Create client in roster (status: pending)
  const client = await clientsService.create({
    name: submission.clientData.name || 'New Lead',
    email: submission.clientData.email,
    phone: submission.clientData.phone,
    status: 'pending',
    engagementTypeId: null,  // To be set
    
    // Store tool results in client metrics
    metrics: mapToolResultsToMetrics(submission.results),
    
    // Track source
    referralSource: 'lead_gen_tool',
    leadGenToolId: submission.toolId,
  });
  
  // 3. Update submission with client ID
  await toolSubmissionsService.update(saved.id, {
    clientId: client.id,
    status: 'invited'
  });
  
  // 4. Show success modal with next steps
  return {
    client,
    submission: saved,
    nextSteps: ['send_invite', 'set_engagement_type', 'view_profile']
  };
}
```

```typescript
export interface LeadGenService {
  // Tool configuration
  getTools(): Promise<LeadGenTool[]>;
  getToolConfig(toolId: string, mentorId: string): Promise<MentorToolConfig | null>;
  updateToolConfig(toolId: string, mentorId: string, config: Partial<MentorToolConfig>): Promise<MentorToolConfig>;
  createCustomTool(tool: Omit<LeadGenTool, 'id' | 'createdAt' | 'updatedAt'>): Promise<LeadGenTool>;
  deleteCustomTool(toolId: string): Promise<void>;
  
  // Submissions
  createSubmission(submission: Omit<ToolSubmission, 'id'>): Promise<ToolSubmission>;
  getSubmissions(mentorId: string, filters?: { toolId?: string; status?: string }): Promise<ToolSubmission[]>;
  updateSubmission(id: string, updates: Partial<ToolSubmission>): Promise<ToolSubmission>;
  
  // Stats
  getToolStats(mentorId: string, toolId: string): Promise<{ views: number; completions: number; signups: number }>;
}
```
