# HiPat - Mentor Directives Complete Redesign

## The Core Problem with Current Design

**Current:** Directives are just canned messages with triggers
**Better:** Directives are **INSTRUCTIONS to Pat about what to DO**

A directive should be like telling an assistant:
> "Hey Pat, after every workout, I want you to summarize what they did, compare it to their last session, highlight any improvements, and if they're slacking, give them a gentle nudge."

NOT just:
> "Send this message when workout completes"

---

## Part 1: Conceptual Redesign

### What a Directive Really Is

A **Mentor Directive** is an instruction set that tells Pat:
1. **WHEN** to activate (trigger)
2. **WHAT** to analyze/gather (data points)
3. **WHAT ACTION** to take (analyze, summarize, compare, alert, encourage, remind)
4. **WHO** to send it to (client, mentor, or both)
5. **HOW** to deliver it (tone, format, urgency)

### Directive Types

| Type | Description | Example |
|------|-------------|---------|
| **Analysis** | Pat analyzes data and provides insights | "After each workout, compare to last session and highlight improvements" |
| **Summary** | Pat summarizes activity over a period | "Every Sunday, send weekly progress summary" |
| **Alert** | Pat notifies when conditions are met | "Alert me if a client misses 3 workouts in a row" |
| **Reminder** | Pat reminds client to do something | "Remind to log water if none logged by 2 PM" |
| **Encouragement** | Pat sends motivation based on achievements | "Celebrate workout streaks at 7, 14, 21, 30 days" |
| **Check-in** | Pat asks client how they're feeling | "After rest days, ask how recovery is going" |
| **Coaching Cue** | Pat delivers specific training advice | "Before leg day, remind about proper squat form" |

## Part 2: New Data Model

### Updated MentorDirective Type

```typescript
export interface MentorDirective {
  id: string;
  mentorId: string;
  
  // Basic Info
  name: string;
  description?: string; // What this directive does in plain English
  
  // Assignment
  assignmentType: 'all' | 'group' | 'individual';
  clientId?: string;
  groupId?: string;
  
  // Directive Type
  directiveType: 'analysis' | 'summary' | 'alert' | 'reminder' | 'encouragement' | 'check_in' | 'coaching_cue';
  
  // WHEN - Trigger Configuration
  trigger: DirectiveTrigger;
  
  // WHAT - Data Points to Include
  dataPoints: DataPointConfig[];
  
  // ACTION - What Pat Should Do
  action: DirectiveAction;
  
  // WHO - Recipients
  recipients: {
    sendToClient: boolean;
    sendToMentor: boolean;
  };
  
  // HOW - Delivery Settings
  delivery: {
    tone: 'encouraging' | 'neutral' | 'direct' | 'celebratory';
    urgency: 'low' | 'medium' | 'high';
    format: 'brief' | 'detailed' | 'bullet_points';
  };
  
  // Custom Message (optional - if mentor wants specific wording)
  customMessage?: string;
  
  // Settings
  isActive: boolean;
  category: 'workout' | 'nutrition' | 'recovery' | 'motivation' | 'general';
  
  // Stats
  triggeredCount: number;
  effectivenessScore?: number;
  lastTriggered?: Date;
  
  createdAt: Date;
  updatedAt: Date;
}

export interface DirectiveTrigger {
  // Event-based trigger
  event?: TriggerEvent;
  eventConditions?: Record<string, any>;
  
  // Time-based trigger
  schedule?: {
    type: 'daily' | 'weekly' | 'monthly';
    time?: string; // "14:00"
    days?: ('mon' | 'tue' | 'wed' | 'thu' | 'fri' | 'sat' | 'sun')[];
    dayOfMonth?: number;
  };
  
  // Condition-based trigger
  condition?: {
    metric: string;
    operator: 'above' | 'below' | 'equals' | 'missing_for';
    value: number;
    unit?: string;
  };
}

export type TriggerEvent = 
  | 'workout_completed'
  | 'workout_missed'
  | 'meal_logged'
  | 'day_end'
  | 'week_end'
  | 'streak_milestone'
  | 'weight_logged'
  | 'goal_achieved'
  | 'check_in_time'
  | 'rest_day'
  | 'before_workout'
  | 'app_opened'
  | 'inactive_period';

export interface DataPointConfig {
  dataPoint: DataPoint;
  comparison?: 'previous' | 'average' | 'goal' | 'best';
  timeframe?: 'today' | 'this_week' | 'this_month' | 'last_7_days' | 'last_30_days';
}

export type DataPoint =
  | 'workout_summary'
  | 'workout_volume'
  | 'workout_duration'
  | 'exercises_completed'
  | 'intensity_rating'
  | 'calories_burned'
  | 'protein_intake'
  | 'calorie_intake'
  | 'water_intake'
  | 'sleep_hours'
  | 'sleep_quality'
  | 'weight'
  | 'body_measurements'
  | 'streak_count'
  | 'weekly_compliance'
  | 'mood_rating'
  | 'energy_level'
  | 'progress_photos'
  | 'personal_records';

export interface DirectiveAction {
  actionType: 'analyze' | 'summarize' | 'compare' | 'alert' | 'remind' | 'encourage' | 'ask' | 'deliver_message';
  
  // For analyze/summarize
  analysisInstructions?: string; // "Focus on progressive overload and form improvements"
  
  // For compare
  compareWith?: 'last_session' | 'last_week' | 'baseline' | 'goal';
  highlightImprovements?: boolean;
  highlightConcerns?: boolean;
  
  // For ask (check-in)
  questions?: string[]; // ["How are you feeling?", "Any soreness?"]
  
  // For alert
  alertCondition?: string;
}
```

export interface ClientGroup {
