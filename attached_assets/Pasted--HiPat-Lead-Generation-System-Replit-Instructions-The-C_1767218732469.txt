# HiPat Lead Generation System - Replit Instructions

## The Core Concept

**The Onboarding Tools (TDEE Calculator, Workout Assessment) ARE the account creation mechanism.**

They're not just assessments - they're lead generation tools that:
1. Deliver value upfront (the results)
2. Create a HiPat account automatically
3. Link the new user to whoever referred them
4. Track for revenue share

async function createAccountFromLeadGen(data) {
  
  // 1. Check if email exists
  const existing = await findByEmail(data.email);
  if (existing) {
    // Don't create duplicate - link to referrer instead
    return handleExistingUser(existing, data.referrerId);
  }
  
  // 2. Create account
  const tempPassword = generateTempPassword();  // e.g., "Xk9$mP2n"
  const user = await createUser({
    email: data.email,
    firstName: data.firstName,
    lastName: data.lastName,
    phone: data.phone,
    password: hash(tempPassword),
    requiresPasswordChange: true,
    subscriptionTier: 'free',
  });
  
  // 3. Attach assessment results (if any)
  if (data.assessmentResults) {
    await saveAssessmentToProfile(user.id, data.assessmentResults);
    // Pre-populate: TDEE, macros, goals, etc.
  }
  
  // 4. Create referral link
  await createReferralRecord({
    newUserId: user.id,
    referrerId: data.referrerId,
    referrerType: data.referrerType,  // 'mentor' or 'circle_user'
    source: data.source,
    toolUsed: data.toolUsed,
    revShareStatus: 'pending',
  });
  
  // 5. Send welcome email
  await sendEmail(user.email, {
    template: data.assessmentResults ? 'welcome_with_results' : 'welcome_basic',
    data: {
      firstName: user.firstName,
      tempPassword: tempPassword,
      assessmentResults: data.assessmentResults,
      referrerName: await getReferrerName(data.referrerId),
      webAppUrl: 'https://app.hipat.com/login',
      iosAppUrl: 'https://apps.apple.com/app/hipat/...',
      androidAppUrl: 'https://play.google.com/store/apps/details?id=...',
    }
  });
  
  // 6. Notify referrer
  await notifyReferrer(data.referrerId, {
    type: 'new_signup',
    userName: `${user.firstName} ${user.lastName}`,
    source: data.source,
  });
  
  return user;
}
